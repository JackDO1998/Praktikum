\section{Fehler}
\label{sec:Fehler}
Der Mittelwert:
\begin{center}
  \begin{equation}
    \label{eq:Mittelwert}
  \bar{x} = \frac{1}{n} \sum \nolimits_{i=0} x_i
  \end{equation} 
\end{center}

Die Standardabweichung:
\begin{center}
  \begin{equation}
    \label{eq:standardabweichung}
    \sigma=\sqrt{\frac{\sum(x_i-\bar{x})^2}{n-1}}
  \end{equation}
\end{center}

Der Fehler des Mittelwertes:
\begin{center}
  \begin{equation}
    \label{eq:mittelwertfehler}
    \sigma_{\bar{x}}=\frac{\sigma}{\sqrt{n}}
  \end{equation}

  
\end{center}
